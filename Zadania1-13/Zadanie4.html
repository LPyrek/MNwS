Za pomocą odpowiednich symulacji zbadać wpływ liczby klas (na które dzielone są dane) odsetek odrzuceń prawdziwej hipotezy głównej (tzn. błąd I rodzaju) w teście 𝜒 2 w przypadku weryfikacji zgodności z rozkładem 𝜒 2 o różnych stopniach swobody, tzn. generować dane pochodzące z rozkładu 𝜒 2 o 𝑘 stopniach swobody i weryfikować hipotezę, że dane mają rozkład 𝜒 2 o 𝑘 st. swobody. Uzyskane wyniki należy przedstawić na odpowiednich wykresach ilustrujących rozmiary testów z uwzględnieniem:
- liczby danych,
- liczby klas,
- liczby stopni swobody generowanego rozkładu.

library(ggplot2)
library(zoo)

power_fun <- function(n,df,no_of_bins){
  mean(
    sapply(1:100, function(i){
      sample <- pchisq(rchisq(n,df),df)
      min_ <- min(sample)
      max_ <- max(sample)
      bin_width <- round((max_-min_)/no_of_bins,10)
      h <- hist(sample,breaks=c(seq(min_,max_+bin_width,bin_width)), plot = FALSE)
      breaks_cdf <- pchisq(h$breaks,0,1)
      probs <- rollapply(breaks_cdf, 2, function(x) x[2] - x[1])
      chisq.test(h$counts, p=probs, rescale.p=TRUE)$p.value < 0.05
    }))
}
bins_seq = c(2,3,5,10,25,50)
n_seq = c(seq(5,100,5),seq(125,300,25))
df_seq = c(3,5,10,25,50)

for(bins in bins_seq){
  data_list <- NULL
  data <- NULL
  data_list <-
    lapply(n_seq, function(n){
      data.frame(
        n = n,
        df = df_seq,
        power = sapply(df_seq, function(df) power_fun(n, df, bins))
      )  
    })
  data <- do.call(rbind, data_list)
  plot <- ggplot(data = data, aes(x=n,y=power, col = factor(df))) +
    geom_line() + 
    geom_hline(yintercept=0.05, linetype='dotted') + 
    geom_hline(yintercept=1, linetype='dotted') +
    ggtitle(paste("liczba klas:",bins)) + labs(col = "Stopnie swobody", y = "error")
  print(plot)
}
