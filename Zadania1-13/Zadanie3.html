library(ggplot2)
power_fun <- function(n, df1, df2){
  mean(
    sapply(1:1000, function(i) {
      sample1 = (rchisq(n, df1) - df1)/sqrt(2*df1)
      sample2 = (rchisq(n, df2) - df2)/sqrt(2*df2)
      ks.test(sample1,sample2)$p.val < 0.05
    })
  )
}
